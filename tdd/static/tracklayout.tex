\documentclass[tikz]{standalone}

\tikzset{
    points/.pic = {
	\foreach \p in {(0,4),(1,3),(2,4),(3,4),(3,3),(3,2),(0,1),(0,3),(2,1),(2,2),(3, 1),(6, 4),(4, 4),(6, 2),(5, 1),(4, 3)} \fill \p circle [radius=0.1];
    },
    tracks/.pic = {
	\foreach \x in {0, 1,...,6} \draw (\x, 0) -- (\x, 5);
    },
    between/.pic = {
	\pic{tracks};
	\draw[green] (0, 4) -- (1, 3) -- (3, 4) -- (2, 4);
	\draw[red] (3,3) -- (3, 2);
    },
    xcrossing/.pic = {
	\pic{between};
	\draw[red] (0,1) -- (2,2) (0,3) -- (2,1);
    },
    okcrossing/.pic = {
	\pic{xcrossing};
	\draw[green] (3, 1) -- (6, 4) -- (4, 4) -- (6, 2) (5, 1) -- (4, 3);
    }
}

\begin{document}
\foreach\picture in {tracks, between, xcrossing, okcrossing} {
    \begin{tikzpicture}[every pic/.style={scale=1.3,line width=2}]
    \pic{\picture};
    \pic{points};
\end{tikzpicture}
}
\end{document}
